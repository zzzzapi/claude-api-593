# OpenAI API中转调用示例
import requests

API_KEY = "your_4z_api_key"  # 替换为您的API密钥
API_URL = "https://api.zzzzapi.com/v1/chat/completions"  # 中转API地址

    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

data = {
    "model": "gpt-4o",  # 支持gpt-3.5-turbo、gpt-4、gpt-4o等所有OpenAI模型
    "messages": [
        {"role": "system", "content": "你是一位AI助手，专长于解答编程问题。"},
        {"role": "user", "content": "请用Python实现一个简单的网络爬虫"}
    ],
    "temperature": 0.7,
    "max_tokens": 2000
}

response = requests.post(API_URL, headers=headers, json=data)
print(response.json()['choices'][0]['message']['content'])

Node.js示例


// OpenAI API中转调用示例 (Node.js)
const axios = require('axios');

const API_KEY = 'your_4z_api_key';
const API_URL = 'https://api.zzzzapi.com/v1/chat/completions';

async function callOpenAI() {
  try {
    const response = await axios.post(API_URL, 
      {
        model: 'gpt-4o',
        messages: [
          {role: 'system', content: '你是一位AI助手，专长于解答编程问题。'},
          {role: 'user', content: '请用JavaScript实现一个简单的网络爬虫'}
        ],
        temperature: 0.7,
        max_tokens: 2000
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        }
      }
    );
    
    console.log(response.data.choices[0].message.content);
  } catch (error) {
    console.error('Error:', error.response ? error.response.data : error.message);
  }
}

callOpenAI();
